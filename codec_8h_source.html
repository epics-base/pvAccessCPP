76  int oldval;
77  // epicsAtomic doesn't have unconditional swap

79  oldval = epics::atomic::get(val);
80  } while(epics::atomic::compareAndSwap(val, oldval, newval)!=oldval);

82  }

84  return epics::atomic::get(val);
85  }
